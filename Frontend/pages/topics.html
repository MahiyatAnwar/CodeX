Tips & Tricks

Two Pointer Technique

Use two pointers to traverse the array from both ends for problems like palindrome checking or pair searching.

left = 0, right = len(arr) - 1
Sliding Window

Maintain a window that slides through the array for subarray/substring problems with fixed or dynamic size.

window_start = 0, window_end = 0
Character Frequency

Use hash maps to count character frequencies for anagram and permutation problems.

freq_map = defaultdict(int)
String Builder

Use StringBuilder (Java) or list joining (Python) for efficient string concatenation.

result = []
result.append(char)
return ''.join(result)

Find Maximum Subarray Sum (Kadane's Algorithm)
FUNCTION maxSubarraySum(arr):
    max_so_far = arr[0]
    current_max = arr[0]
    
    FOR i FROM 1 TO length(arr) - 1:
        current_max = MAX(arr[i], current_max + arr[i])
        max_so_far = MAX(max_so_far, current_max)
    
    RETURN max_so_far
END FUNCTION

def reverse_string(s):
    left, right = 0, len(s) - 1
    chars = list(s)
    
    while left < right:
        chars[left], chars[right] = chars[right], chars[left]
        left += 1
        right -= 1
    
    return ''.join(chars)

def max_subarray_sum(arr):

public class ArraySolutions {
    public static String reverseString(String str) {
        char[] chars = str.toCharArray();
        int left = 0, right = chars.length - 1;
        
        while (left < right) {
            char temp = chars[left];
            chars[left] = chars[right];
            chars[right] = temp;
            left++;
            right--;
        }
        
        return new String(chars);
    }
    
    public static int maxSubarraySum(int[] arr) {
        int maxSoFar = arr[0];
        int currentMax = arr[0];
        
        for (int i = 1; i < arr.length; i++) {
            currentMax = Math.max(arr[i], currentMax + arr[i]);
            maxSoFar = Math.max(maxSoFar, currentMax);
        }
        
        return maxSoFar;
    }
}
